```tsx
import { useEffect, useRef, useState } from 'react';
import {
  AlertTriangle,
  CheckCircle2,
  Clock,
  Settings,
  VolumeX,
  Volume2,
  TrendingUp,
  TrendingDown,
  Minus,
} from 'lucide-react';
import { useDashboardStore } from '../store';

interface O2AnalyzerCardV2Props {
  title: string;
  o2Level: number;
  alarmLevel: number;
  isAlarmActive: boolean;
  lastCalibration: string;
  onSettingsClick?: () => void;
  onMuteAlarm?: () => void;
  isMuted?: boolean;
  showAlarm?: boolean;
}

export function O2AnalyzerCardV2({
  title,
  o2Level,
  alarmLevel,
  isAlarmActive,
  lastCalibration,
  onSettingsClick,
  onMuteAlarm,
  isMuted = false,
  showAlarm = true,
}: O2AnalyzerCardV2Props) {
  const darkMode = useDashboardStore((s) => s.darkMode);
  const prevO2Ref = useRef<number>(o2Level);
  const [trend, setTrend] = useState<'up' | 'down' | 'stable'>('stable');

  const isHighOxygen = o2Level > alarmLevel;
  const hasAlarm = showAlarm && (isAlarmActive || isHighOxygen);

  // Trend arrow logic: compare with previous value (updated every 30s via polling)
  useEffect(() => {
    const timer = setTimeout(() => {
      const diff = o2Level - prevO2Ref.current;
      if (diff > 0.2) setTrend('up');
      else if (diff < -0.2) setTrend('down');
      else setTrend('stable');
      prevO2Ref.current = o2Level;
    }, 30000); // Compare every 30 seconds
    return () => clearTimeout(timer);
  }, [o2Level]);

  // Card classes
  const glassCard = darkMode
    ? 'bg-white/5 backdrop-blur-xl border border-white/10 shadow-[0_8px_32px_rgba(0,0,0,0.3)]'
    : 'bg-white/60 backdrop-blur-xl border border-white/80 shadow-[0_8px_32px_rgba(31,38,135,0.15)]';

  const alarmGlow = hasAlarm
    ? darkMode
      ? 'shadow-[0_0_60px_rgba(239,68,68,0.3)] border-red-500/80 bg-red-500/8'
      : 'shadow-[0_0_40px_rgba(239,68,68,0.2)] border-red-500/80 bg-red-500/6'
    : '';

  const alarmPulse = hasAlarm && !isMuted ? 'animate-alarm-pulse' : '';
  const alarmGlowAnim = hasAlarm && !isMuted ? 'animate-alarm-glow' : '';

  // Value color
  const valueColor = hasAlarm
    ? 'bg-gradient-to-b from-red-500 to-rose-400 bg-clip-text text-transparent'
    : 'bg-gradient-to-b from-emerald-400 to-emerald-300 bg-clip-text text-transparent';

  // Trend icon
  const TrendIcon = trend === 'up' ? TrendingUp : trend === 'down' ? TrendingDown : Minus;
  const trendColor = trend === 'up' ? 'text-rose-400' : trend === 'down' ? 'text-emerald-400' : darkMode ? 'text-slate-500' : 'text-slate-400';

  return (
    <div className="relative">
      {/* Alarm Banner */}
      {hasAlarm && (
        <div
          className={`flex items-center justify-between px-5 py-2 rounded-t-3xl ${
            isMuted
              ? 'bg-slate-600'
              : 'animate-alarm-flash bg-red-600'
          }`}
        >
          <div className="flex items-center gap-2">
            <AlertTriangle className="w-5 h-5 text-white" />
            <span className="text-white font-bold text-sm tracking-wide">
              {isMuted ? 'ALARM MUTED' : 'HIGH O2 ALARM'}
            </span>
          </div>
          {onMuteAlarm && (
            <button
              onClick={onMuteAlarm}
              className="p-1.5 rounded-full bg-white/20 hover:bg-white/30 transition-colors"
              aria-label={isMuted ? 'Unmute alarm' : 'Mute alarm'}
            >
              {isMuted ? (
                <VolumeX className="w-4 h-4 text-white" />
              ) : (
                <Volume2 className="w-4 h-4 text-white" />
              )}
            </button>
          )}
        </div>
      )}

      {/* Main Card */}
      <div
        className={`
          ${showAlarm ? 'w-[480px] h-[480px]' : 'w-full h-auto'}
          ${hasAlarm ? 'rounded-b-3xl' : 'rounded-3xl'}
          ${glassCard}
          ${alarmGlow}
          ${alarmPulse}
          ${alarmGlowAnim}
          p-6 flex flex-col justify-between
          transition-all duration-500
          animate-fade-up
        `}
      >
        {/* Header: Title + Settings */}
        <div className="flex items-center justify-between">
          <h2 className={`${showAlarm ? 'text-4xl' : 'text-2xl'} font-bold ${darkMode ? 'text-white' : 'text-blue-600'}`}>
            {title}
          </h2>
          {onSettingsClick && (
            <button
              onClick={onSettingsClick}
              className={`p-2.5 rounded-full transition-all duration-200 hover:scale-110 ${
                darkMode
                  ? 'bg-white/10 hover:bg-white/20'
                  : 'bg-white/80 hover:bg-white shadow-lg'
              }`}
              aria-label="Settings"
            >
              <Settings className={`w-5 h-5 ${darkMode ? 'text-blue-400' : 'text-blue-600'}`} />
            </button>
          )}
        </div>

        {/* O2 Level */}
        <div className="text-center flex-1 flex flex-col items-center justify-center">
          <span className={`text-sm font-medium tracking-wider uppercase ${darkMode ? 'text-slate-400' : 'text-slate-500'}`}>
            O2 Level
          </span>
          <div className="flex items-center justify-center gap-3 mt-2">
            <span
              className={`${showAlarm ? 'text-8xl' : 'text-6xl'} font-bold tabular-nums transition-all duration-500 ${valueColor}`}
            >
              {o2Level.toFixed(1)}%
            </span>
            <TrendIcon className={`${showAlarm ? 'w-8 h-8' : 'w-6 h-6'} ${trendColor} animate-fade-in`} />
          </div>
        </div>

        {/* Alarm Info Panel */}
        {showAlarm && (
          <div className={`rounded-2xl p-4 mb-3 ${darkMode ? 'bg-white/5' : 'bg-white/50'}`}>
            <div className="flex items-center justify-between mb-2">
              <span className={`text-sm font-medium ${darkMode ? 'text-slate-400' : 'text-slate-600'}`}>
                High Alarm:
              </span>
              <span className={`text-lg font-bold ${darkMode ? 'text-blue-400' : 'text-blue-600'}`}>
                {alarmLevel}%
              </span>
            </div>
            <div className="flex items-center justify-between">
              <span className={`text-sm font-medium ${darkMode ? 'text-slate-400' : 'text-slate-600'}`}>
                Status:
              </span>
              <div className="flex items-center gap-2">
                {hasAlarm ? (
                  <>
                    <AlertTriangle className="w-5 h-5 text-red-500" />
                    <span className="text-xs font-bold px-2.5 py-1 rounded-full bg-red-500/20 text-red-400">
                      HIGH O2
                    </span>
                  </>
                ) : (
                  <>
                    <CheckCircle2 className="w-5 h-5 text-emerald-500" />
                    <span className="text-xs font-bold px-2.5 py-1 rounded-full bg-emerald-500/20 text-emerald-400">
                      NORMAL
                    </span>
                  </>
                )}
              </div>
            </div>
          </div>
        )}

        {/* Last Calibration */}
        <div className={`rounded-xl p-3 ${darkMode ? 'bg-white/5' : 'bg-white/30'}`}>
          <div className="flex items-center gap-3">
            <Clock className={`w-4 h-4 ${darkMode ? 'text-blue-400' : 'text-blue-600'}`} />
            <div>
              <div className={`text-xs font-medium ${darkMode ? 'text-slate-400' : 'text-slate-600'}`}>
                Last Calibration
              </div>
              <div className={`text-sm ${darkMode ? 'text-slate-300' : 'text-slate-700'}`}>
                {lastCalibration}
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
```

**Step 1: Create the new page**

Create `renderer/pages/o2-analyzer-v2.tsx`. This is based on the existing `o2-analyzer.tsx` but replaces the layout and card component:

```tsx
import React, { useEffect, useState, useRef, useCallback } from 'react';
import Head from 'next/head';
import toast, { Toaster } from 'react-hot-toast';
import io from 'socket.io-client';
import { Sun, Moon } from 'lucide-react';
import { O2AnalyzerCardV2 } from '../components/O2AnalyzerCardV2';
import { O2AnalyzerSettings } from '../components/O2AnalyzerSettings';
import { useChambers } from '../hooks/useChambers';
import { useLatestReading } from '../hooks/useReadings';
import { useChamberAlarms } from '../hooks/useAlarms';
import { Chamber } from '../api/chambers';
import { useDashboardStore } from '../store';

// ChamberCard wrapper — same logic as v1, uses V2 card component
interface ChamberCardProps {
  chamber: Chamber;
  onSettingsClick: () => void;
  onMuteAlarm: () => void;
  isMuted: boolean;
  onAlarmStateChange: (chamberId: number, hasAlarm: boolean) => void;
  showAlarm?: boolean;
}

const ChamberCard: React.FC<ChamberCardProps> = ({
  chamber,
  onSettingsClick,
  onMuteAlarm,
  isMuted,
  onAlarmStateChange,
  showAlarm = true,
}) => {
  const { reading } = useLatestReading(chamber.id, true, 5000);
  const { alarms } = useChamberAlarms(chamber.id);

  const o2Level = reading?.o2Level ?? chamber.lastValue ?? 21.0;
  const alarmLevel = chamber.alarmLevelHigh;

  const activeAlarms = Array.isArray(alarms) ? alarms.filter((a) => a.isActive) : [];
  const hasBackendAlarms = activeAlarms.length > 0;
  const hasO2Alarm = o2Level > alarmLevel;
  const hasActiveAlarms = showAlarm ? (hasBackendAlarms || hasO2Alarm) : false;

  useEffect(() => {
    onAlarmStateChange(chamber.id, hasActiveAlarms);
  }, [chamber.id, hasActiveAlarms, onAlarmStateChange]);

  const lastCalibration = chamber.calibrationDate
    ? new Date(chamber.calibrationDate).toLocaleDateString('en-US') +
      ' ' +
      new Date(chamber.calibrationDate).toLocaleTimeString('en-US', {
        hour: '2-digit',
        minute: '2-digit',
      })
    : 'Never';

  const displayName = !showAlarm
    ? chamber.name.replace(/^fio(\d+)$/i, 'FiO₂ - $1')
    : chamber.name;

  return (
    <O2AnalyzerCardV2
      title={displayName}
      o2Level={o2Level}
      alarmLevel={alarmLevel}
      isAlarmActive={hasActiveAlarms}
      lastCalibration={lastCalibration}
      onSettingsClick={onSettingsClick}
      onMuteAlarm={showAlarm ? onMuteAlarm : undefined}
      isMuted={isMuted}
      showAlarm={showAlarm}
    />
  );
};

export default function O2AnalyzerV2Page() {
  const [currentTime, setCurrentTime] = useState('');
  const [currentDate, setCurrentDate] = useState('');
  const [settingsOpen, setSettingsOpen] = useState(false);
  const [selectedChamber, setSelectedChamber] = useState<Chamber | null>(null);
  const [mutedAlarms, setMutedAlarms] = useState<{ [key: number]: boolean }>({});

  const darkMode = useDashboardStore((s) => s.darkMode);
  const setDarkMode = useDashboardStore((s) => s.setDarkMode);

  const socketRef = useRef<any>(null);
  const prevAlarmStatesRef = useRef<{ [key: number]: boolean }>({});

  const { chambers, loading: chambersLoading, error: chambersError, refetch: refetchChambers } = useChambers();
  const mainAnteChambers = chambers.filter((c) => !c.name.toLowerCase().startsWith('fio'));

  // Socket connection
  useEffect(() => {
    const socket = io('http://192.168.77.100:4000', {
      transports: ['websocket', 'polling'],
      reconnection: true,
      reconnectionAttempts: 5,
      reconnectionDelay: 1000,
    });
    socket.on('connect', () => console.log('O2v2: Connected'));
    socket.on('disconnect', (r) => console.log('O2v2: Disconnected', r));
    socketRef.current = socket;
    return () => { socket.disconnect(); };
  }, []);

  // Clock
  useEffect(() => {
    const update = () => {
      const now = new Date();
      setCurrentDate(
        `${now.getDate().toString().padStart(2, '0')}.${(now.getMonth() + 1).toString().padStart(2, '0')}.${now.getFullYear()}`
      );
      setCurrentTime(
        `${now.getHours().toString().padStart(2, '0')}:${now.getMinutes().toString().padStart(2, '0')}`
      );
    };
    update();
    const interval = setInterval(update, 1000);
    return () => clearInterval(interval);
  }, []);

  // Error toast
  useEffect(() => {
    if (chambersError) toast.error(`Error: ${chambersError}`, { duration: 5000, position: 'top-center' });
  }, [chambersError]);

  const handleSettingsClick = (chamber: Chamber) => {
    setSelectedChamber(chamber);
    setSettingsOpen(true);
  };

  // Mute alarm — same PLC logic as v1
  const handleMuteAlarm = (chamber: Chamber) => {
    const isMuted = mutedAlarms[chamber.id] || false;
    const isMainChamber = chamber.name.toLowerCase().includes('main');
    const register = isMainChamber ? 'M0407' : 'M0408';

    if (isMuted) {
      setMutedAlarms((prev) => ({ ...prev, [chamber.id]: false }));
      toast.success(`${chamber.name} alarm activated`, { duration: 3000, position: 'top-center' });
      if (socketRef.current?.connected) socketRef.current.emit('writeBit', { register, value: 1 });
    } else {
      setMutedAlarms((prev) => ({ ...prev, [chamber.id]: true }));
      toast.success(`${chamber.name} alarm muted for 5 minutes`, { duration: 3000, position: 'top-center' });
      if (socketRef.current?.connected) socketRef.current.emit('writeBit', { register, value: 0 });
      setTimeout(() => {
        setMutedAlarms((prev) => ({ ...prev, [chamber.id]: false }));
        if (socketRef.current?.connected) socketRef.current.emit('writeBit', { register, value: 1 });
      }, 5 * 60 * 1000);
    }
  };

  // Alarm state change → PLC write — same logic as v1
  const handleAlarmStateChange = useCallback(
    (chamberId: number, hasAlarm: boolean) => {
      const prevState = prevAlarmStatesRef.current[chamberId];
      if (prevState === undefined) {
        prevAlarmStatesRef.current[chamberId] = hasAlarm;
        return;
      }
      if (prevState !== hasAlarm) {
        prevAlarmStatesRef.current[chamberId] = hasAlarm;
        if (hasAlarm && socketRef.current?.connected) {
          const chamber = chambers.find((c) => c.id === chamberId);
          if (chamber) {
            const register = chamber.name.toLowerCase().includes('main') ? 'M0407' : 'M0408';
            socketRef.current.emit('writeBit', { register, value: 1 });
          }
        }
      }
    },
    [chambers]
  );

  // Loading
  if (chambersLoading) {
    return (
      <div className={`h-screen flex items-center justify-center ${darkMode ? 'bg-[#1a1a2e]' : 'bg-[#e8edf5]'}`}>
        <div className="animate-spin rounded-full h-16 w-16 border-b-2 border-blue-400" />
      </div>
    );
  }

  // Error
  if (chambersError && chambers.length === 0) {
    return (
      <div className={`h-screen flex items-center justify-center ${darkMode ? 'bg-[#1a1a2e]' : 'bg-[#e8edf5]'}`}>
        <p className="text-xl text-red-500">{chambersError}</p>
      </div>
    );
  }

  return (
    <>
      <Head>
        <title>O2 Analyzer - Hipertech</title>
      </Head>

      <div
        className={`h-screen flex flex-col transition-all duration-500 ${
          darkMode
            ? 'bg-gradient-to-br from-[#1a1a2e] via-[#16213e] to-[#0f3460] animate-gradient-shift'
            : 'bg-gradient-to-br from-[#e8edf5] to-[#f0f4ff]'
        }`}
      >
        {/* Header */}
        <div className="flex justify-between items-center py-3 px-8">
          <img
            alt="Hipertech Logo"
            src="/external/hipertechlogo2501-ygje.svg"
            className={`h-12 ${darkMode ? 'brightness-0 invert' : ''}`}
          />

          {/* Theme Toggle */}
          <button
            onClick={() => setDarkMode(!darkMode)}
            className={`p-2.5 rounded-full transition-all duration-300 hover:scale-110 ${
              darkMode ? 'bg-white/10 hover:bg-white/20' : 'bg-white/80 hover:bg-white shadow-md'
            }`}
            aria-label="Toggle theme"
          >
            {darkMode ? (
              <Sun className="w-5 h-5 text-amber-400" />
            ) : (
              <Moon className="w-5 h-5 text-slate-600" />
            )}
          </button>

          <img
            alt="O2 Monitor Logo"
            src="/external/O2Monitor.png"
            className={`h-12 ${darkMode ? 'brightness-0 invert' : ''}`}
          />
        </div>

        {/* Main Content */}
        <div className="flex-1 flex items-center justify-center px-8">
          <div className="flex gap-8 items-start">
            {mainAnteChambers.map((chamber) => (
              <ChamberCard
                key={chamber.id}
                chamber={chamber}
                onSettingsClick={() => handleSettingsClick(chamber)}
                onMuteAlarm={() => handleMuteAlarm(chamber)}
                isMuted={mutedAlarms[chamber.id] || false}
                onAlarmStateChange={handleAlarmStateChange}
                showAlarm={true}
              />
            ))}
          </div>
        </div>

        {/* Footer */}
        <div className="flex justify-end items-center px-8 py-3">
          <span className={`font-bold text-lg ${darkMode ? 'text-blue-300' : 'text-blue-600'}`}>
            {currentDate} - {currentTime}
          </span>
        </div>

        {/* Settings Modal */}
        {selectedChamber && (
          <O2AnalyzerSettings
            isOpen={settingsOpen}
            onClose={() => { setSettingsOpen(false); setSelectedChamber(null); }}
            chamber={selectedChamber}
            onUpdate={refetchChambers}
          />
        )}
      </div>

      <Toaster
        toastOptions={{
          style: { background: '#363636', color: '#fff', fontSize: '16px', fontWeight: '500' },
          success: { iconTheme: { primary: '#22c55e', secondary: '#fff' } },
          error: { iconTheme: { primary: '#ef4444', secondary: '#fff' } },
        }}
      />
    </>
  );
}
```
